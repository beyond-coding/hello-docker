*** development stage ***

-- step 1 - test
mvn test package
-- step 2 - dockerize
mvn clean package
docker build -t beyondcoding/hello-docker .
docker login -e="$DOCKER_EMAIL" -u="$DOCKER_USERNAME" -p="$DOCKER_PASSWORD"
docker push beyondcoding/hello-docker
-- step 3 - run
docker pull beyondcoding/hello-docker
docker run -d -p 8080:8080 beyondcoding/hello-docker
sleep 30
curl --fail http://localhost:8080


*** test/production stage ***

-- step 1 - test
mvn test package
-- step 2 - dockerize
mvn clean package
PROJECT_VERSION=`mvn org.apache.maven.plugins:maven-help-plugin:2.1.1:evaluate -Dexpression=project.version|grep -Ev '(^\[|Download\w+:)'`
docker build -t beyondcoding/hello-docker:$PROJECT_VERSION .
docker login -e="$DOCKER_EMAIL" -u="$DOCKER_USERNAME" -p="$DOCKER_PASSWORD"
docker push beyondcoding/hello-docker:$PROJECT_VERSION
-- step 3 - run
PROJECT_VERSION=`mvn org.apache.maven.plugins:maven-help-plugin:2.1.1:evaluate -Dexpression=project.version|grep -Ev '(^\[|Download\w+:)'`
docker pull beyondcoding/hello-docker:$PROJECT_VERSION
docker run -d -p 8080:8080 beyondcoding/hello-docker:$PROJECT_VERSION
sleep 30
curl --fail http://localhost:8080
-- step 4 deploy
mvn clean package
PROJECT_VERSION=`mvn org.apache.maven.plugins:maven-help-plugin:2.1.1:evaluate -Dexpression=project.version|grep -Ev '(^\[|Download\w+:)'`
ssh -o BatchMode=yes -o StrictHostKeyChecking=no  gherrero@dev.pixtri.com 'echo hola'
scp target/hello-docker-$PROJECT_VERSION.jar gherrero@dev.pixtri.com:/continous-deployment/.
